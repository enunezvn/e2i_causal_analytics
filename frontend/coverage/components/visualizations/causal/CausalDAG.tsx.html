function positionNodes(

  edges: CausalEdge[],
  width: number,
  height: number,
  nodeRadius: number
): PositionedNode[] {
  const depths = calculateNodeDepths(nodes, edges);
  const levels = new Map<number, CausalNode[]>();
 
  nodes.forEach((n) => {
    const depth = depths.get(n.id) ?? 0;
    if (!levels.has(depth)) {
      levels.set(depth, []);
    }
    levels.get(depth)!.push(n);
  });

  const levelHeight = height / (maxDepth + 2);
