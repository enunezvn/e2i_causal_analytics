üóÑÔ∏è SQL Migration: 007_causal_validation_tables.sql

-- ============================================ -- Migration: 007_causal_validation_tables.sql -- Purpose: Add causal validation infrastructure -- ============================================ -- 1. Create ENUMs CREATE TYPE refutation_test_type AS ENUM ( 'placebo_treatment', 'random_common_cause', 'data_subset', 'bootstrap', 'sensitivity_e_value' ); CREATE TYPE validation_status AS ENUM ( 'passed', 'failed', 'warning', 'skipped' ); CREATE TYPE gate_decision AS ENUM ( 'proceed', 'review', 'block' ); CREATE TYPE expert_review_type AS ENUM ( 'dag_approval', 'methodology_review', 'quarterly_audit', 'ad_hoc_validation' ); -- 2. Create causal_validations table CREATE TABLE causal_validations ( validation_id UUID PRIMARY KEY DEFAULT gen_random_uuid(), estimate_id UUID NOT NULL, -- FK to causal_paths or ml_experiments test_type refutation_test_type NOT NULL, status validation_status NOT NULL, original_effect DECIMAL(10,6), refuted_effect DECIMAL(10,6), p_value DECIMAL(5,4), delta_percent DECIMAL(5,2), confidence_score DECIMAL(3,2) CHECK (confidence_score >= 0 AND confidence_score <= 1), gate_decision gate_decision NOT NULL DEFAULT 'review', test_config JSONB, -- Test parameters used details_json JSONB, -- Full refutation output agent_activity_id UUID, -- FK to agent_activities data_split VARCHAR(20), -- ML split tracking created_at TIMESTAMPTZ DEFAULT now(), CONSTRAINT fk_agent_activity FOREIGN KEY (agent_activity_id) REFERENCES agent_activities(activity_id) ); -- 3. Create expert_reviews table CREATE TABLE expert_reviews ( review_id UUID PRIMARY KEY DEFAULT gen_random_uuid(), review_type expert_review_type NOT NULL, dag_version_hash VARCHAR(64), -- SHA256 of DAG structure reviewer_id VARCHAR(100) NOT NULL, reviewer_role VARCHAR(50), -- 'commercial_ops', 'medical_affairs', etc. approval_status VARCHAR(20) NOT NULL DEFAULT 'pending', checklist_json JSONB, -- Completed checklist items comments_json JSONB, -- Reviewer notes brand VARCHAR(50), analysis_context TEXT, -- What analysis this review covers valid_until DATE, -- Expiration for quarterly reviews created_at TIMESTAMPTZ DEFAULT now(), approved_at TIMESTAMPTZ ); -- 4. Create indexes CREATE INDEX idx_validations_estimate ON causal_validations(estimate_id); CREATE INDEX idx_validations_status ON causal_validations(status); CREATE INDEX idx_validations_gate ON causal_validations(gate_decision); CREATE INDEX idx_validations_created ON causal_validations(created_at DESC); CREATE INDEX idx_expert_reviews_dag ON expert_reviews(dag_version_hash); CREATE INDEX idx_expert_reviews_status ON expert_reviews(approval_status); -- 5. Create validation summary view CREATE VIEW v_validation_summary AS SELECT estimate_id, COUNT(*) as total_tests, COUNT(*) FILTER (WHERE status = 'passed') as passed_count, COUNT(*) FILTER (WHERE status = 'failed') as failed_count, AVG(confidence_score) as avg_confidence, MAX(gate_decision) as final_gate -- 'block' > 'review' > 'proceed' FROM causal_validations GROUP BY estimate_id;
